
public class Main {

    // Function to count number of items in a list, using recursion.
    public static int countRecursively(int[] arr, int index) {
        // If the array is empty, return 0
        if (index == arr.length) {
            return 0;
        }

        // Return 1 plus the count of the rest of the array
        return 1 + countRecursively(arr, index + 1);
    }

    public static void main(String[] args) {
        // Test cases
        int[] emptyArray = {};
        int[] array123 = {1, 2, 3};

        // Print results of test cases
        System.out.println(countRecursively(emptyArray, 0)); // Expected output: 0
        System.out.println(countRecursively(array123, 0)); // Expected output: 3
    }
